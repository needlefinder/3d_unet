def plot_loss(x, maxiter=400, label=''):
    loss = np.array(x.Scalars('loss_val'))
    loss_t = loss[:,0][:maxiter] - loss[0,0]
    loss_v = loss[:,2][:maxiter]
    loss_t2 = []
    for i in range(len(loss_t)):
        loss_t2.append(i)
    
    plt.plot(loss_t2,loss_v, label=label)
